const initialValues = {
  email:'',
  password:''
}

const validate=values => {
  let errors = {}

if (!values.password) {
 errors.password = 'Required'
}

if (!values.email) {


  errors.email = 'This field is required'
} else if (!/^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,4}$/i.test(values.email)) {
 errors.email = 'Invalid email format'
}

return errors
}

const onSubmit = values => {
 // const [isValid, setIsValid] = useState(false);
 // const [focusMessage, setMessage] = useState(""); 
async function staticFetcher(){

  try{
    const loginInfo = {
      identifier:values.email,
      password: values.password
  }

 


  const login = await fetch(API_URL+"/auth/local", {
    method: "POST",
    headers: {
        'Accept': 'application/json',
        'Content-Type': 'application/json'
    },
    body: JSON.stringify(loginInfo)
})
const loginResponse = await login.json();
console.log("Response",login.status);

// Set
nookies.set(null, 'jwt', loginResponse.jwt ,{
  maxAge: 30 * 24 * 60 * 60,
  path: '/',
})

if (login.status==200){
 // setIsValid(true)
  console.log("should of gone to admin")
 // setMessage("Your details look good you will log on in 2 sec")
  Router.push('/admin')
}
}
catch{
  //setIsValid(false)
//  setMessage("Something is not right! Check your credentials carefully")
  console.log("error")
}
finally{
  Router.push('/login')
}
}
staticFetcher()

}

function FormLogin() {
 
  const formik = useFormik({
    initialValues,
    onSubmit,
    validate,
  })
 
 // const [isValid, setIsValid] = useState(false);
 // const [focusMessage, setMessage] = useState(""); 

  return (
    <>
    <Head>
        <title>Login | AIDA</title>
      </Head>
      <Box
        component="main"
        sx={{
          alignItems: 'center',
          display: 'flex',
          flexGrow: 1,
          minHeight: '100%',
        }}
        className="height-75"
      >
        <Container    sx={{
          backgroundColor:'white',
        }} maxWidth="sm">
          <NextLink
            href="/"
            passHref
          >
            <Button
              component="a"
              startIcon={<ArrowBackIcon fontSize="small" />}
            >
              Dashboard
            </Button>
          </NextLink>
    <div>
      <form onSubmit={formik.handleSubmit}>
      <div className='form-control'>
              <label htmlFor='email'>email</label>
              <input
               type='text' 
               id='email'
                name='email' 
                onChange={formik.handleChange}
                onBlur={formik.handleBlur}
                 value ={formik.values.email}/>
        </div>
        {formik.touched.email && formik.errors.email ? (
          <div className='error'>{formik.errors.email}</div>
        ) : null}

            <div className='form-control'>
              <label htmlFor='password'>password</label>
              <input type='password' id='password' name='password' onChange={formik.handleChange} onBlur={formik.handleBlur} value ={formik.values.password} />
            </div>
            {formik.touched.password && formik.errors.password ? (
          <div className='error'>{formik.errors.password}</div>
        ) : null}

            <Button className="form-button"  type='submit' >Submit</Button>
      </form>
    </div>
    </Container>
    </Box>
    </>
  )
}

export default FormLogin;
